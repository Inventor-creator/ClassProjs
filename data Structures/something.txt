	userId INT PRIMARY KEY,
	DOB VARCHAR(8),
	userName VARCHAR(20),
	passwords VARCHAR(20)

);

CREATE TABLE Playlists(
	playListId INT ,
	playlistName VARCHAR(20),
	creationDate VARCHAR(8),
	userId INT,
	FOREIGN KEY (userId ) REFERENCES Users(userId)

);
